import pandas as pd

num_rows = 900
columns = ["Store ID", "Store Area", "Available items", "Customer_count_daily", "Store_Sales"]

np.random.seed(42)
store_ids =np.arange(1, num_rows + 1)

store_sales =  np.random.uniform(1000, 50000, size = num_rows)          # Store sales in currency

# Creating a DataFrame

df = pd.DataFrame({
